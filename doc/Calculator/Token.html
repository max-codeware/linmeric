def initialize(value)
  @val = value
  if OP.include? value then
    @tag = :OPERATOR
  elsif value.number? then
    @tag = :NUMBER
    @val = value.to_n
  elsif value == "(" then
    @tag = :L_PAR
  elsif value == ")" then
    @tag = :R_PAR
  end
end
