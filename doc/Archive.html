def next_
  if  @i < (@myArray.size - 1) 
    @i += 1
    return @myArray[@i].clone
  else
    @i = @myArray.size
    return ""
  end
  @i = @myArray.size
  return ""
end

def previous
  if not @i == 0 then
    @i -= 1
    return @myArray[@i].clone
  else
    @i = 0
    return ""
  end
  @i = 0
  return ""
end

def store(str)
  if @myArray.size == 20
    @myArray = @myArray[1...@myArray.size] + [str]
    # @i = @myArray.size
  else
    @myArray[@myArray.size] = str
    @i = @myArray.size
  end
end
